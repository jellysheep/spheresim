PROJECT(Circles_OpenClGl)
cmake_minimum_required(VERSION 2.8)

FIND_PACKAGE(Qt4 REQUIRED)
FIND_PACKAGE(OpenGL REQUIRED)
SET(CMAKE_REQUIRED_LIBRARIES ${GL_LIBRARY})
LINK_DIRECTORIES(${OPENGL_LIBRARIES})
INCLUDE_DIRECTORIES(${OPENGL_INCLUDE_DIR})

SET(QT_USE_QTOPENGL TRUE)

SET(Circles_SOURCES main.cpp GLWidget.cpp ClTimer.cpp)
SET(Circles_HEADERS GLWidget.h ClTimer.h CirclesHolder.h)

QT4_WRAP_CPP(Circles_HEADERS_MOC ${Circles_HEADERS})

INCLUDE(${QT_USE_FILE})
ADD_DEFINITIONS(${QT_DEFINITIONS})

ADD_EXECUTABLE(circles_ocgl ${Circles_SOURCES} 
	${Circles_HEADERS_MOC})
TARGET_LINK_LIBRARIES(circles_ocgl ${QT_LIBRARIES} ${OPENGL_LIBRARIES})
